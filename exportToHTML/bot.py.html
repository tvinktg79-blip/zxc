# ========= НАСТРОЙКИ НАКАЗАНИЙ / АНТИФЛУДА / ЯЗЫКА =========

WARN_FOR_MUTE = 3       # после 3 нарушений — мут
WARN_FOR_BAN = 6        # после 6 — бан
MUTE_SECONDS = 600      # длительность мута (10 минут)

FLOOD_MAX_MSG = 10       # максимум сообщений
FLOOD_INTERVAL = 30      # за 30 секунд

# фильтр языка
MIN_LETTERS_FOR_LANG_CHECK = 5   # минимум букв, чтобы проверять язык
MIN_CYR_RATIO = 0.5               # минимум доли кириллицы (20 %)

bot = Bot(API_TOKEN)
dp = Dispatcher()

# user_id -> количество нарушений
violations = defaultdict(int)

# chat_id -> последние сообщения (message_id, text, user_id)
LAST_MESSAGES = defaultdict(lambda: deque(maxlen=200))

# антифлуд: user_id -> deque[timestamps]
flood_messages = defaultdict(lambda: deque(maxlen=50))

# статистика: user_id -> удалено сообщений
stats_deleted = defaultdict(int)

REPLACE_MAP = str.maketrans({
    "@": "a",
    "4": "a",
    "0": "o",
    "3": "e",
    "1": "i",
    "$": "s",
    "€": "е",
})

LATIN_TO_CYR = str.maketrans({
    "a": "а",
    "b": "б",
    "c": "с",
    "e": "е",
    "h": "н",
    "k": "к",
    "m": "м",
    "o": "о",
    "p": "р",
    "r": "р",
    "t": "т",
    "x": "х",
    "y": "у",
})


def normalize_text(text: str) -> str:
    t = text.lower()
    t = t.translate(REPLACE_MAP)
    t = t.translate(LATIN_TO_CYR)
    return t


def has_bad_word(text: str) -> bool:
    t = normalize_text(text)
    for w in BAD_WORDS_EXACT:
        if w in t:
            return True
    for root in BAD_WORD_PATTERNS:
        if root in t:
            return True
    return False


def is_suspicious_language(text: str) -> bool:
    """True, если в длинном сообщении слишком мало кириллицы (подозрительно для русского чата)."""
    letters = [ch for ch in text if ch.isalpha()]
    if len(letters) < MIN_LETTERS_FOR_LANG_CHECK:
        return False  # слишком короткое, не трогаем
    total = len(letters)
    cyr = sum("а" <= ch.lower() <= "я" or ch.lower() == "ё" for ch in letters)
    ratio = cyr / total
    return ratio < MIN_CYR_RATIO


async def punish_user(chat_id: int, user_id: int, count: int):
    # бан
    if count >= WARN_FOR_BAN:
        try:
            await bot.ban_chat_member(chat_id, user_id)
            logging.info(f"Пользователь {user_id} забанен (нарушений: {count})")
        except Exception as e:
            logging.error(f"Не удалось забанить пользователя {user_id}: {e}")
        return

    # мут
    if count >= WARN_FOR_MUTE:
        until_date = int(time.time()) + MUTE_SECONDS
        perms = ChatPermissions(can_send_messages=False)
        try:
            await bot.restrict_chat_member(
                chat_id,
                user_id,
                permissions=perms,
                until_date=until_date,
            )
            logging.info(f"Пользователь {user_id} замучен на {MUTE_SECONDS} сек (нарушений: {count})")
        except Exception as e:
            logging.error(f"Не удалось замутить пользователя {user_id}: {e}")


def check_flood(user_id: int) -> bool:
    now = time.time()
    q = flood_messages[user_id]
    q.append(now)
    while q and now - q[0] > FLOOD_INTERVAL:
        q.popleft()
    return len(q) > FLOOD_MAX_MSG

@dp.message(F.text.startswith("/"))
async def log_commands(message: Message):
    logging.info(f"[CMD] chat={message.chat.id} user={message.from_user.id}: {message.text!r}")


@dp.message(F.text & ~F.text.startswith("/"))
async def filter_bad_words(message: Message):
    logging.info(f"[TXT] chat={message.chat.id} user={message.from_user.id}: {message.text!r}")
    chat_id = message.chat.id
    user_id = message.from_user.id

    # фильтр языка / кракозябр
    if is_suspicious_language(message.text):
        try:
            await message.delete()
        except Exception as e:
            logging.error(f"Не удалось удалить сообщение по языковому фильтру: {e}")
        warn = (
            f"{message.from_user.mention_html()}, сообщение удалено: "
            f"используйте, пожалуйста, русский язык."
        )
        await bot.send_message(chat_id, warn, parse_mode="HTML")
        return

    # антифлуд
    if check_flood(user_id):
        try:
            await message.delete()
        except Exception as e:
            logging.error(f"Не удалось удалить флуд-сообщение: {e}")

        until_date = int(time.time()) + MUTE_SECONDS
        perms = ChatPermissions(can_send_messages=False)
        try:
            await bot.restrict_chat_member(
                chat_id,
                user_id,
                permissions=perms,
                until_date=until_date,
            )
            warn = (
                f"{message.from_user.mention_html()}, не флуди.\n"
                f"Ты замучен на {MUTE_SECONDS} секунд за флуд."
            )
        except Exception as e:
            logging.error(f"Не удалось замутить за флуд пользователя {user_id}: {e}")
            warn = f"{message.from_user.mention_html()}, не флуди (слишком много сообщений)."

        await bot.send_message(chat_id, warn, parse_mode="HTML")
        return

    # сохраняем в буфер для периодической проверки
    LAST_MESSAGES[chat_id].append((message.message_id, message.text, user_id))

    # онлайн-проверка мата
    if not has_bad_word(message.text):
        return

    violations[user_id] += 1
    stats_deleted[user_id] += 1
    count = violations[user_id]

    try:
        await message.delete()
    except Exception as e:
        logging.error(f"Не удалось удалить сообщение: {e}")

    warn_text = (
        f"{message.from_user.mention_html()}, без мата.\n"
        f"Нарушение {count}."
    )
    await bot.send_message(chat_id, warn_text, parse_mode="HTML")

    await punish_user(chat_id, user_id, count)

async def is_admin(message: Message) -> bool:
    try:
        member = await bot.get_chat_member(message.chat.id, message.from_user.id)
        return member.is_chat_admin()
    except Exception as e:
        logging.error(f"Не удалось проверить админа: {e}")
        return False


@dp.message(Command("addbad"))
async def cmd_addbad(message: Message):
    if not await is_admin(message):
        return
    parts = message.text.split(maxsplit=1)
    if len(parts) < 2:
        await message.reply("Использование: /addbad слово")
        return
    word = parts[1].strip().lower()
    if not word:
        await message.reply("Пустое слово добавить нельзя.")
        return
    BAD_WORDS_EXACT.add(word)
    await message.reply(f"Слово «{word}» добавлено в список мата.")


@dp.message(Command("delbad"))
async def cmd_delbad(message: Message):
    if not await is_admin(message):
        return
    parts = message.text.split(maxsplit=1)
    if len(parts) < 2:
        await message.reply("Использование: /delbad слово")
        return
    word = parts[1].strip().lower()

    removed = False
    if word in BAD_WORDS_EXACT:
        BAD_WORDS_EXACT.remove(word)
        removed = True
    if word in BAD_WORD_PATTERNS:
        BAD_WORD_PATTERNS.remove(word)
        removed = True

    if removed:
        await message.reply(f"Слово «{word}» удалено из списков мата.")
    else:
        await message.reply(f"Слова «{word}» нет в списках.")


@dp.message(Command("stats"))
async def cmd_stats(message: Message):
    if not await is_admin(message):
        return
    if not stats_deleted:
        await message.reply("Пока нет статистики нарушений.")
        return
    lines = []
    for user_id, cnt in sorted(stats_deleted.items(), key=lambda x: x[1], reverse=True)[:10]:
        lines.append(f"ID {user_id}: удалено {cnt} сообщений")
    text = "Топ нарушителей:\n" + "\n".join(lines)
    await message.reply(text)

        user = event.new_chat_member.user

        text = (

        await bot.send_message(chat_id, text, parse_mode="HTML")


# ========= ПЕРИОДИЧЕСКАЯ ПРОВЕРКА =========

async def periodic_cleanup():
    while True:
        await asyncio.sleep(3600)
        logging.info("Запуск периодической проверки чатов на мат")

        for chat_id, messages in list(LAST_MESSAGES.items()):
            for msg_id, text, user_id in list(messages):
                if not has_bad_word(text):
                    continue
                try:
                    await bot.delete_message(chat_id, msg_id)
                    stats_deleted[user_id] += 1
                    violations[user_id] += 1
                    count = violations[user_id]
                    warn_text = (
                        f"Сообщение с матом удалено при периодической проверке.\n"
                        f"Нарушение {count}."
                    )
                    await bot.send_message(chat_id, warn_text)
                    try:
                        messages.remove((msg_id, text, user_id))
                    except ValueError:
                        pass
                    await punish_user(chat_id, user_id, count)
                except Exception as e:
                    logging.error(f"Не удалось удалить старое сообщение {msg_id}: {e}")


async def main():
    while True:
        try:
            logging.info("Старт polling")
            asyncio.create_task(periodic_cleanup())
            await dp.start_polling(bot)
        except Exception as e:
            logging.error(f"Polling упал: {e}")
            logging.info("Ждём 5 секунд и пробуем переподключиться...")
            await asyncio.sleep(5)


if __name__ == "__main__":
    asyncio.run(main())
